//Cascade 1 and 4

vec4 ProjectLC(vec2 luvp, float lProbeSize, vec3 probePos, vec3 probeDir, vec3 probeTan,
               vec3 probeBit, vec3 probeNor, vec2 probeBin, vec3 pPos, mat3 pEyeMat,
               vec2 cascadeOffset, vec4 rayInfo, float probeWSize, bool lFrame, float frame, float lCascade) {
    //Last cascade projection
    vec2 lProbeGridOffset = ProbeStochOffset(frame + 1., lCascade, lProbeSize);
    if (lFrame) lProbeGridOffset = ProbeStochOffset(frame, lCascade, lProbeSize);
    vec4 lProbeAttr = texture(iChannel0, (luvp + lProbeGridOffset + vec2(((lFrame) ? HRES.x : 0.), 0.))*IRES);
    vec3 lProbeNor = normalize(FloatToVec3(lProbeAttr.z)*2. - 1.);
    if (lProbeAttr.w < -5. || dot(probeDir, lProbeNor) <= 0.) return vec4(0.);
    vec3 lProbePos = pPos + normalize(vec3(((luvp + lProbeGridOffset)*IHRES*2. - 1.)*ASPECT, 1.)*pEyeMat)*lProbeAttr.w;
    vec3 lProbeBit; vec3 lProbeTan = TBN(lProbeNor, lProbeBit);
    mat3 lProbeMat = TBN(lProbeNor);
    vec2 luvd = ProjectDir(vec3(dot(probeDir, lProbeBit), dot(probeDir, lProbeTan), dot(probeDir, lProbeNor)), lProbeSize);
    vec2 luvdc = clamp(luvd, vec2(2.5), vec2(lProbeSize - 2.5));
    vec4 sampleBin = vec4(0.);
    for (float x = -2.; x < 2.5; x++) {
        for (float y = -2.; y < 2.5; y++) {
            vec2 sLUV = cascadeOffset + luvp + luvdc + vec2(x, y);
            vec4 sLD = texture(iChannel1, sLUV*IRES);
            vec3 sRelPos = lProbePos + (ComputeDir(luvdc + vec2(x, y), lProbeSize)*lProbeMat)*sLD.w - probePos;
            vec3 sRPosTBN = vec3(dot(sRelPos, probeBit), dot(sRelPos, probeTan), dot(sRelPos, probeNor));
            vec2 sProjectedBin = ProjectDir(sRPosTBN, lProbeSize*0.5);
            if (sProjectedBin.x < -0.5 || DFBox(sProjectedBin - probeBin, vec2(1.)) > 0.) continue;
            if (rayInfo.w < probeWSize && rayInfo.w < dot(probeDir, sRelPos) - 0.005) continue;
            sampleBin += vec4(max(vec3(0.), sLD.xyz), 1.);
        }
    }
    float w = max(0.001, (dot(lProbeNor, probeNor) - 0.8)/0.2*float(abs(dot(lProbePos - probePos, probeNor)) < 0.02));
    if (sampleBin.w > 0.5) return vec4(sampleBin.xyz/sampleBin.w*w, w);
    else return vec4(0.);
}

void mainImage(out vec4 fragColor, in vec2 fragCoord) {
    vec4 info = vec4(-20.);
    vec3 sunDir = GetSunDirection(iTime);
    float probeCascade = 1. + floor(fragCoord.y*IHRES.y)*3.;
    float probeSize = pow(2., probeCascade + 1.);
    vec2 probePositions = floor(HRES/probeSize);
    vec2 probeSSRes = probePositions*probeSize;
    vec2 modUV = mod(fragCoord, HRES);
    vec2 probeUV = mod(modUV, probeSize);
    
    if (fragCoord.x < HRES.x && DFBox(mod(fragCoord, HRES), probeSSRes) < 0.) {
        //Probes
        vec4 cMouse = texture(iChannel0, vec2(HRES.x + 0.5, HRES.y + 0.5)*IRES);
        if (fragCoord.y < HRES.y) cMouse = texture(iChannel0, vec2(HRES.x + 1.5, HRES.y + 0.5)*IRES);
        vec2 probeGridOffset = ProbeStochOffset(float(iFrame) + 1., probeCascade, probeSize);
        if (fragCoord.y < HRES.y) probeGridOffset = ProbeStochOffset(float(iFrame), probeCascade, probeSize);
        vec2 probeDirUV = floor(modUV/probeSize)*probeSize + probeGridOffset;
        vec3 pPos = GetCameraPos(cMouse, IRES);
        vec3 pEye = GetCameraEye(cMouse, IRES);
        mat3 pEyeTBN = TBN(pEye);
        vec3 pDir = normalize(vec3((probeDirUV*IHRES*2. - 1.)*ASPECT, 1.)*pEyeTBN);
        vec4 pAttr = texture(iChannel0, (probeDirUV + vec2(((fragCoord.y < HRES.y) ? HRES.x : 0.), 0.))*IRES);
        if (pAttr.w > -15.) {
            //Probe is valid
            vec3 probeNor = normalize(FloatToVec3(pAttr.z)*2. - 1.);
            vec3 probePos = pPos + pDir*pAttr.w + probeNor*0.0001;
            vec3 probeBit; vec3 probeTan = TBN(probeNor, probeBit);
            vec3 probeDir = ComputeDir(probeUV, probeSize)*TBN(probeNor);
            
            //Trace ray
            HIT rayHit = TraceRay(probePos, probeDir, 1000000., iTime);
            if (rayHit.n.x > -15.) {
                if (rayHit.c.x > 1.) {
                    info = vec4(rayHit.c, rayHit.t);
                } else {
                    info = vec4(0., 0., 0., rayHit.t);
                    vec3 hitP = probePos + probeDir*rayHit.t + rayHit.n*0.001;
                    vec3 sunLight = GetSunLight(iTime, hitP);
                    if (TraceRay(hitP, sunDir, 10000., iTime).n.x < -15.) {
                        info.xyz += max(0., dot(rayHit.n, sunDir))*sunLight*rayHit.c;
                    }
                }
            } else {
                info = vec4(GetSkyLight(probeDir), 1000000.);
            }
            
            //Merge
            float probeWSize = ProbeWSize(probeDirUV, probeSize, pAttr.w, IRES, ASPECT);
            float lProbeSize = probeSize*2.;
            vec2 lPUVPos = clamp(floor(modUV/lProbeSize - 0.5), vec2(0.), floor(HRES/lProbeSize) - 2.)*lProbeSize;
            vec2 fPUVPos = fract(modUV/lProbeSize - 0.5);
            if (floor(modUV.x/lProbeSize - 0.5) > floor(HRES.x/lProbeSize) - 1.5) fPUVPos.x = 1.;
            else if (floor(modUV.x/lProbeSize - 0.5) < -0.5) fPUVPos.x = 0.;
            if (floor(modUV.y/lProbeSize - 0.5) > floor(HRES.y/lProbeSize) - 1.5) fPUVPos.y = 1.;
            else if (floor(modUV.y/lProbeSize - 0.5) < -0.5) fPUVPos.y = 0.;
            vec2 cascadeOffset = vec2(0., floor(fragCoord.y*IHRES.y)*HRES.y);
            vec4 S0 = ProjectLC(lPUVPos, lProbeSize, probePos,
                                probeDir, probeTan, probeBit, probeNor, probeUV - 0.5,
                                pPos, pEyeTBN, cascadeOffset, info, probeWSize,
                                fragCoord.y < HRES.y, float(iFrame), probeCascade + 1.);
            vec4 S1 = ProjectLC(lPUVPos + vec2(lProbeSize, 0.), lProbeSize, probePos,
                                probeDir, probeTan, probeBit, probeNor, probeUV - 0.5,
                                pPos, pEyeTBN, cascadeOffset, info, probeWSize,
                                fragCoord.y < HRES.y, float(iFrame), probeCascade + 1.);
            vec4 S2 = ProjectLC(lPUVPos + vec2(0., lProbeSize), lProbeSize, probePos,
                                probeDir, probeTan, probeBit, probeNor, probeUV - 0.5,
                                pPos, pEyeTBN, cascadeOffset, info, probeWSize,
                                fragCoord.y < HRES.y, float(iFrame), probeCascade + 1.);
            vec4 S3 = ProjectLC(lPUVPos + lProbeSize, lProbeSize, probePos,
                                probeDir, probeTan, probeBit, probeNor, probeUV - 0.5,
                                pPos, pEyeTBN, cascadeOffset, info, probeWSize,
                                fragCoord.y < HRES.y, float(iFrame), probeCascade + 1.);
            if (S0.w + S1.w + S2.w + S3.w > 0.0005) {
                info.xyz = mix(mix(S0.xyz, S1.xyz, fPUVPos.x), mix(S2.xyz, S3.xyz, fPUVPos.x), fPUVPos.y)/
                           max(0.0001, mix(mix(S0.w, S1.w, fPUVPos.x), mix(S2.w, S3.w, fPUVPos.x), fPUVPos.y));
            }
        }
    }
    fragColor = info;
}
